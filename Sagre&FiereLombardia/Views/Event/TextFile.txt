public IActionResult Index(int pageNumber = 1, string searchQuery = "", string filterComune = "", string selectedComune = "")
{
    int pageSize = 10;

    var query = _dbContext.TuoDbSet.AsQueryable();

    // Applica eventuali filtri
    if (!string.IsNullOrEmpty(searchQuery))
        query = query.Where(x => x.Nome.Contains(searchQuery)); // esempio

    // Paginazione vera e propria
    var totalItems = query.Count();
    var totalPages = (int)Math.Ceiling(totalItems / (double)pageSize);

    var paginatedResults = query
        .Skip((pageNumber - 1) * pageSize)
        .Take(pageSize)
        .ToList();

    // Passa dati alla View (usa ViewModel se possibile)
    ViewBag.pageNumber = pageNumber;
    ViewBag.totalPages = totalPages;
    ViewBag.searchQuery = searchQuery;
    ViewBag.filterComune = filterComune;
    ViewBag.selectedComune = selectedComune;

    return View(paginatedResults);
}
